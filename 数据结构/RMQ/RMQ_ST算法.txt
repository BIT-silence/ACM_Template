// RMQ-STÀ„∑® O(nlogn) - O(1)

#include <iostream>
#include <cstdio>
#include <cstring>

#define MAXN 100010

using namespace std;

int a[MAXN], w[MAXN], d[MAXN][20];

void initST(const int *a, int n) {
    int i, j;

	for (w[0] = -1, i = 1; i <= n; ++i) w[i] = (i & (i - 1)) ? w[i - 1] : w[i - 1] + 1;
    for (i = 0; i < n; ++i) d[i][0] = a[i];
    for (j = 1; (1 << j) <= n; ++j) {
        for (i = 0; i + (1 << j) <= n; ++i) {
            d[i][j] = min(d[i][j - 1], d[i + (1 << (j - 1))][j - 1]);
        }
    }
}

int query(int a, int b) {
    int l = w[b - a + 1];
    return min(d[a][l], d[b - (1 << l) + 1][l]);
}

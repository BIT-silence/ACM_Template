#define MAX 110
#include <iostream>
#include <cstdio>
#include <cstring>

typedef __int64 i64;

using namespace std;

i64 mat[MAX][MAX];

i64 det(i64 a[MAX][MAX], int n, i64 p) {
    int i, j, k;
	i64 t, ans = 1;
	
	for (i = 0; i < n; ++i) {
		for (j = 0; j < n; ++j) {
			a[i][j] %= p;
		}
	}
    for (i = 0; i < n; ++i) {
        for (j = i + 1; j < n; ++j) {
            for ( ; a[j][i]; ans = -ans) {
                for (t = a[i][i] / a[j][i], k = 0; k < n; ++k) {
                    a[i][k] = (a[i][k] - a[j][k] * t) % p;
                    swap(a[i][k], a[j][k]);
                }
            }
        }
        if (!a[i][i]) return 0;
        ans = ans * a[i][i] % p;
    }

    return (ans % p + p) % p;
}
